Supports both TypeScript and JSDoc's syntax

@typedef

Lets us define interfaces

/**
 * @typedef {Object} Car
 * @property {string} brand
 * @property {number} year
 * @property {Object} engine
 * @property {string} engine.type
 * @property {number} engine.power
 */
interface Car {
  brand: string;
  year: number;
  engine: {
    type: string,
    power: number
  }
}
